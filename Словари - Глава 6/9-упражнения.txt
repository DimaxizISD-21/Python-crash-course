glossary = {
	'Массив': 'упорядоченный набор последовательность данных',
	'Функция': 'это поименованный блок программного кода (тело функции)',
	'Класс': 'описание объекта в виде набора его свойств, которые аналогичны обычным переменным и функциям',
	'Операторы': 'предназначены для задания команд, которые последовательно выполняет компьютер',
	'Переменная': 'принимает значение или, другими словами, в переменную записывается значение с помощью оператора присваивания'
}

glossary['Айди'] = '(от англ. «identificator») – идентификатор.'
glossary['Айпи '] = '(от англ. «Internet Protocol») – уникальный адрес. Каждый компьютер имеет свой IP, который предоставляет ему провайдер.'
glossary['Айтишник '] = 'программист, специалист в информационных технологиях (ИТ).'
glossary['Апдейт '] = '(от англ «update») – обновление.'
glossary['Аутсорс'] = '(от англ. «outsourcing») – передача предприятием/компанией выполнение определенной работы не своим работникам, а другим (фрилансерам или работникам других компаний)'


for glossar in glossary.keys():
	print("\n" + glossar.title() + " это " + glossary[glossar])

print("\n")

big_rivers_of_country = {
	'Нил': 'Египете',
	'Амазонка': 'Бразилии',
	'Янцзы': 'Китае'
}

for river in big_rivers_of_country.keys():
	print("Эта река " + river.title() + " находится в " + 
		big_rivers_of_country[river] + "!" )

for river in big_rivers_of_country.keys():
	print("\n" + river.title())

for river in big_rivers_of_country.values():
	print("\n" + river.title())
